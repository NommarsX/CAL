    <script>

      const allowedDates = [

          '2024-09-01',

          '2024-09-03',

          '2024-09-07',

          '2024-09-11',

          '2024-09-17',

          '2024-09-21',

      ]

      const picker = new easepick.create({

        element: document.getElementById('datepicker'),

        css: [

          'https://cdn.jsdelivr.net/npm/@easepick/bundle@1.2.1/dist/index.css',

        ],

        plugins: ['LockPlugin'],

        LockPlugin: {

          filter(date, picked) {

            return !allowedDates.includes(date.format('YYYY-MM-DD'));

          },

        },

      });

    </script>
